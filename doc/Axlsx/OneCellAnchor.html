def initialize(drawing, options={})
  @drawing = drawing
  @width = 0
  @height = 0
  drawing.anchors << self      
  @from = Marker.new
  options.each do |o|
    self.send("#{o[0]}=", o[1]) if self.respond_to? "#{o[0]}="
  end
  @object = Pic.new(self, options)
end

def to_xml(xml)
  xml.send('xdr:oneCellAnchor') {
    xml.send('xdr:from') {
      from.to_xml(xml)
    }
    xml.send('xdr:ext', ext)
    @object.to_xml(xml)
    xml.send('xdr:clientData')
  }
end
