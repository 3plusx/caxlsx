def initialize(chart, options={})
  self.chart = chart
  @chart.series << self
  options.each do |o|
    self.send("#{o[0]}=", o[1]) if self.respond_to? "#{o[0]}="
  end
end

def to_xml(xml)
  xml.send('c:ser') {
    xml.send('c:idx', :val=>index)
    xml.send('c:order', :val=>order || index)
    xml.send('c:tx') {
      xml.send('c:v', self.title)
    }
    yield xml if block_given?
  }
end
