def to_xml()
  builder = Nokogiri::XML::Builder.new(:encoding => ENCODING) do |xml|
    xml.Relationships(:xmlns => Axlsx::RELS_R) {
      each_with_index { |rel, index| rel.to_xml(xml, "rId#{index+1}") }
    }
  end
  builder.to_xml
end
