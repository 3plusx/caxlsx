def to_xml(xml)
  xml.xf(self.instance_values.reject { |k, v| [:alignment, :protection, :extList, :name].include? k.to_sym}) {
    alignment.to_xml(xml) if self.alignment
    protection.to_xml(xml) if self.protection
  }
end
