def to_xml()
  builder = Nokogiri::XML::Builder.new(:encoding => ENCODING) do |xml|
    xml.Types(:xmlns => Axlsx::XML_NS_T) {
      each { |type| type.to_xml(xml) }
    }
  end
  builder.to_xml
end
