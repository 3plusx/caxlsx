def initialize(options={})
  options[:type] ||= :linear
  options.each do |o|
    self.send("#{o[0]}=", o[1]) if self.respond_to? o[0]
  end
  @stop = SimpleTypedList.new GradientStop
end

def to_xml(xml)
  xml.gradientFill(self.instance_values.reject { |k,v| k.to_sym == :stop }) {
    @stop.each { |s| s.to_xml(xml) }
  }
end
