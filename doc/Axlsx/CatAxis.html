def to_xml(xml)
  xml.send('c:catAx') {
    super(xml)
    xml.send('c:auto', :val=>@auto)
    xml.send('c:lblAlgn', :val=>@lblAlgn)
    xml.send('c:lblOffset', :val=>@lblOffset)                 
  }
end
