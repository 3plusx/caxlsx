def initialize(frame, options={})
  super(frame, options)
  @series_type = PieSeries
  @view3D = View3D.new({:rotX=>30, :perspective=>30}.merge(options))
end

def to_xml
  super() do |xml|
    xml.send('c:pie3DChart') {
      xml.send('c:varyColors', :val=>1)
      @series.each { |ser| ser.to_xml(xml) }
    }                      
  end
end
