def initialize(options={})

    self.send("#{o[0]}=", o[1]) if self.respond_to? o[0]
  end

def to_xml(xml)
  xml.font {
    self.instance_values.each do |k, v|
      v.is_a?(Color) ? v.to_xml(xml) : xml.send(k, {:val => v})            
    end
  }
end
