def initialize(options={})
  raise ArgumentError, "Extension and ContentType are required" unless options[:Extension] && options[:ContentType]
  options.each do |o|
    self.send("#{o[0]}=", o[1]) if self.respond_to? "#{o[0]}="
  end      
end
