def to_xml(xml)
  xml.send('c:serAx') {
    super(xml)
    xml.send('c:tickLblSkip', :val=>@tickLblSkip) unless @tickLblSkip.nil?
    xml.send('c:tickMarkSkip', :val=>@tickMarkSkip) unless @tickMarkSkip.nil?
  }
end
